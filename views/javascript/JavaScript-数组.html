                                  Array.prototype.uniq = function(){
  var arr = [], me = this, len = me.length;
  for (var i = 0; i < len; i++) {
    for (var j = i + 1; j < len; j++) {
      if (me[i] === me[j]) j = ++i;
    }
    arr.push(me[i]);
  }
  return arr;
  }

                                  # 累计

                                  语法：reduce(callback[, initVal])

                                  • callback(oldVal, newVal, index, args){}：回调
                                    • oldVal：旧值、累加值
                                    • newVal：新值
                                    • index：新值索引
                                    • args：数组
                                  • initVal：初始值
                                  // 数组前5个元素求和
var arr = [1, 2, 3, 4, 5];

var sum = arr.reduce(function(val, newVal) {
  return val + newVal;
}, 0);

console.log(sum); // => 15
