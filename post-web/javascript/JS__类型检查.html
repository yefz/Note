        # 封装检测方法

        // 封装 toString 方法
if (!Object.prototype.getType) {
  Object.prototype.getType = function() {
    var me = this;
    if(arguments.length == 1) me = arguments[0];
    var type = Object.prototype.toString.call(me);
    return type.slice(8, -1).toLowerCase();
  }
}
