                    # 合并 DOM 插入

                    var listNode = document.getElementById('list')

// 插入 10 个 li 标签
var frag = document.createDocumentFragment();
var x, li;
for(x = 0; x < 10; x++) {
  li = document.createElement('li');
  li.innerHTML = 'List item' + x;
  frag.appendChild(li);
}
// 一次 DOM 插入
listNode.appendChild(frag)
